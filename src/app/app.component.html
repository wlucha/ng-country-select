Installation


ng add @wlucha/ng-country-select

1. Outline Appearance


<ng-country-select
  appearance="outline"
></ng-country-select>

2. Exclude countries


<ng-country-select
  [excludeCountries]="['us', 'de', 'fr']"
></ng-country-select>

3. Include countries


<ng-country-select
  [includeCountries]="['us', 'de', 'fr']"
></ng-country-select>

4. Set Default Country (non-FormControl)


<ng-country-select
  [defaultCountry]="presetCountry"
></ng-country-select>

5. Set Default Country (FormControl)


<ng-country-select
  [formControl]="countryControl"
></ng-country-select>

7. Custom Language Support


import { provideCountrySelectConfig } from '@wlucha/ng-country-select';

export const appConfig: ApplicationConfig = {
  providers: [
    provideCountrySelectConfig({
      extraTranslations: {
        de: { pl: 'Niemcy' },
        at: { pl: 'Austria' },
        pl: { pl: 'Polska' },
        us: { pl: 'Stany Zjednoczone' },
        fr: { pl: 'Francja' },
        // ... more countries
      }
    })
  ]
};
          

<ng-country-select
  [lang]="'pl'"
  [placeholder]="'Szukaj kraju'"
></ng-country-select>

8. Show Alpha2/Alpha3 Codes


<ng-country-select
  [showCodes]="true"
></ng-country-select>

9. Hide Flag


<ng-country-select
  [showFlag]="false"
></ng-country-select>

10. Show Alpha2


<ng-country-select
  [alpha2Only]="true"
></ng-country-select>

11. Set a country programmatically by Alpha2


 <ng-country-select
    [selectedCountryByAlpha2]="alpha2Country"
  ></ng-country-select>

12. Set a country programmatically by Alpha3


  <ng-country-select
      [selectedCountryByAlpha3]="alpha3Country"
    ></ng-country-select>

13. Using required fields


  <ng-country-select
      [required]="true"
      [requiredErrorMessage]="'Please select a country'"
      [showRequiredErrorMessage]="true"
    ></ng-country-select>
